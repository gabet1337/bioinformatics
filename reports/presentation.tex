\chapter*{Our folding approach}
\subsection*{Bruteforce for $n < 20$}
\begin{algorithm}[H]
\caption{\textsc{Bruteforce}}
\begin{algorithmic}[1]
	\FORALL{possible foldings $x$}
		\IF{ $x$ is valid folding}
			\STATE{get score of folding}
			\IF{score is the best}
				\STATE{store the path and score}
			\ENDIF
		\ENDIF
	\ENDFOR
\end{algorithmic}
\end{algorithm}

\subsection*{Sampling for $n \geq 20$}
\begin{algorithm}[H]
\caption{\textsc{Sampling with DFS}}
\begin{algorithmic}[1]
	\WHILE{long time is not long anymore}
		\STATE{generate valid random folding using DFS}
		\STATE{get score of folding}
		\IF{score is the best}
			\STATE{store the path and score}
		\ENDIF
	\ENDWHILE
\end{algorithmic}
\end{algorithm}
